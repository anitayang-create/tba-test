        const apiKey = ""; 


        async function fetchWithRetry(url, options, maxRetries = 5) {

            const delays = [1000, 2000, 4000, 8000, 16000];

            for (let i = 0; i < maxRetries; i++) {

                try {

                    const res = await fetch(url, options);

                    if (!res.ok) throw new Error(`HTTP Error: ${res.status}`);

                    return await res.json();

                } catch (e) {

                    if (i === maxRetries - 1) throw e;

                    await new Promise(r => setTimeout(r, delays[i]));

                }

            }

        }


        async function openAIEvaluation(itemName, category) {

            document.getElementById('ai-modal').classList.add('show');

            document.getElementById('ai-target-item').innerText = itemName;

            document.getElementById('ai-loading').style.display = 'block';

            document.getElementById('ai-content').style.display = 'none';


            if (!apiKey) return document.getElementById('ai-loading').innerHTML = '<span style="color:var(--ant-error)">⚠️ 未設定 API Key。</span>';


            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const sysPrompt = `你是一位專業的藝術電商產品經理。請以 JSON 格式繁體中文評估提案可行性。`;

            const userQuery = `商品名稱：${itemName}\n種類：${category}\n請提供 1. feasibilityScore (1-10) 2. targetAudience (30字) 3. manufacturingAdvice (40字) 4. marketingTagline (一句話)`;


            try {

                const res = await fetchWithRetry(apiUrl, {

                    method: 'POST', headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({

                        contents: [{ parts: [{ text: userQuery }] }],

                        systemInstruction: { parts: [{ text: sysPrompt }] },

                        generationConfig: {

                            responseMimeType: "application/json",

                            responseSchema: {

                                type: "OBJECT",

                                properties: {

                                    feasibilityScore: { type: "INTEGER" }, targetAudience: { type: "STRING" },

                                    manufacturingAdvice: { type: "STRING" }, marketingTagline: { type: "STRING" }

                                },

                                required: ["feasibilityScore", "targetAudience", "manufacturingAdvice", "marketingTagline"]

                            }

                        }

                    })

                });


                let text = res.candidates?.[0]?.content?.parts?.[0]?.text;

                if (!text) throw new Error("API 回應無效");

                text = text.trim().replace(/^`{3}(json)?\n?/i, '').replace(/\n?`{3}$/i, '');

                const data = JSON.parse(text);


                document.getElementById('ai-score').innerHTML = `<span style="font-size:18px;font-weight:600;color:${data.feasibilityScore >= 7 ? 'var(--ant-success)' : 'var(--ant-warning)'}">${data.feasibilityScore} / 10 分</span>`;

                document.getElementById('ai-audience').innerText = data.targetAudience;

                document.getElementById('ai-manufacturing').innerText = data.manufacturingAdvice;

                document.getElementById('ai-marketing').innerText = `「${data.marketingTagline}」`;


                document.getElementById('ai-loading').style.display = 'none';

                document.getElementById('ai-content').style.display = 'block';

            } catch (err) {

                console.error(err);

                document.getElementById('ai-loading').innerHTML = '<span style="color:var(--ant-error)">⚠️ 分析失敗。</span>';

            }

        }


        function closeAIEvaluation() { document.getElementById('ai-modal').classList.remove('show'); }
